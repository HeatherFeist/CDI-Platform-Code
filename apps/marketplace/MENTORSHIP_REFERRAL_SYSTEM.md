## Database Schema

### Referral/Sponsorship System
```sql
-- Referral relationships
CREATE TABLE member_referrals (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  
  sponsor_id UUID REFERENCES auth.users(id) NOT NULL, -- The mentor
  mentee_id UUID REFERENCES auth.users(id) NOT NULL, -- The trainee
  
  -- Status
  status VARCHAR(20) DEFAULT 'training', -- 'training', 'graduated', 'dropped', 'completed'
  
  -- Training Period
  training_started_at TIMESTAMPTZ DEFAULT NOW(),
  training_completed_at TIMESTAMPTZ,
  training_milestones JSONB DEFAULT '[]', -- [{milestone: 'first_sale', completed_at: '2025-10-01'}]
  
  -- Graduation Requirements
  meetings_attended INTEGER DEFAULT 0,
  total_sales INTEGER DEFAULT 0,
  total_revenue DECIMAL(10,2) DEFAULT 0,
  avg_rating DECIMAL(3,2) DEFAULT 0,
  
  -- Accountability
  sponsor_check_ins INTEGER DEFAULT 0, -- How many times sponsor checked in
  last_check_in_at TIMESTAMPTZ,
  mentee_satisfaction_rating INTEGER, -- 1-5 stars for sponsor
  sponsor_notes TEXT,
  
  -- Revenue Sharing
  donation_percentage DECIMAL(5,2) DEFAULT 15.00, -- 10-15%
  donation_allocation VARCHAR(20) DEFAULT 'nonprofit', -- 'sponsor', 'nonprofit', 'split'
  donation_split JSONB, -- {sponsor: 7.5, nonprofit: 7.5} if split
  
  total_donated_to_sponsor DECIMAL(10,2) DEFAULT 0,
  total_donated_to_nonprofit DECIMAL(10,2) DEFAULT 0,
  
  -- Termination
  dropped_at TIMESTAMPTZ,
  dropped_by UUID REFERENCES auth.users(id), -- Who ended it (sponsor or admin)
  drop_reason TEXT,
  
  UNIQUE(sponsor_id, mentee_id)
);

CREATE INDEX idx_member_referrals_sponsor ON member_referrals(sponsor_id);
CREATE INDEX idx_member_referrals_mentee ON member_referrals(mentee_id);
CREATE INDEX idx_member_referrals_status ON member_referrals(status);

-- Donation transactions (tracking)
CREATE TABLE donation_transactions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  
  transaction_id UUID REFERENCES transactions(id) NOT NULL, -- Original sale
  mentee_id UUID REFERENCES auth.users(id) NOT NULL,
  
  -- Recipient
  recipient_type VARCHAR(20) NOT NULL, -- 'sponsor', 'nonprofit'
  recipient_id UUID REFERENCES auth.users(id), -- If sponsor, their user ID
  
  amount DECIMAL(10,2) NOT NULL,
  percentage DECIMAL(5,2) NOT NULL,
  
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_donation_transactions_mentee ON donation_transactions(mentee_id);
CREATE INDEX idx_donation_transactions_recipient ON donation_transactions(recipient_id);
CREATE INDEX idx_donation_transactions_created ON donation_transactions(created_at DESC);

-- Add to users table
ALTER TABLE auth.users ADD COLUMN IF NOT EXISTS can_sponsor BOOLEAN DEFAULT false;
ALTER TABLE auth.users ADD COLUMN IF NOT EXISTS total_mentees INTEGER DEFAULT 0;
ALTER TABLE auth.users ADD COLUMN IF NOT EXISTS successful_graduates INTEGER DEFAULT 0;
ALTER TABLE auth.users ADD COLUMN IF NOT EXISTS mentor_rating DECIMAL(3,2) DEFAULT 0;
ALTER TABLE auth.users ADD COLUMN IF NOT EXISTS total_mentorship_earnings DECIMAL(10,2) DEFAULT 0;
```

### Automatic Donation Processing
```sql
-- Function to process donation after sale
CREATE OR REPLACE FUNCTION process_mentorship_donation()
RETURNS TRIGGER AS $$
DECLARE
  referral_record RECORD;
  donation_amt DECIMAL(10,2);
  sponsor_amt DECIMAL(10,2);
  nonprofit_amt DECIMAL(10,2);
BEGIN
  -- Only process if transaction is completed
  IF NEW.status = 'completed' THEN
    
    -- Get seller's referral relationship
    SELECT * INTO referral_record
    FROM member_referrals
    WHERE mentee_id = (
      SELECT seller_id FROM listings WHERE id = NEW.listing_id
    )
    AND status IN ('training', 'graduated');
    
    IF FOUND THEN
      -- Calculate donation amount
      donation_amt := NEW.total_amount * (referral_record.donation_percentage / 100);
      
      -- Allocate based on status and preference
      IF referral_record.status = 'training' THEN
        -- Training period: 100% to nonprofit
        nonprofit_amt := donation_amt;
        sponsor_amt := 0;
        
        INSERT INTO donation_transactions (
          transaction_id, mentee_id, recipient_type, recipient_id, amount, percentage
        ) VALUES (
          NEW.id, referral_record.mentee_id, 'nonprofit', NULL, nonprofit_amt, referral_record.donation_percentage
        );
        
        -- Update referral record
        UPDATE member_referrals
        SET 
          total_donated_to_nonprofit = total_donated_to_nonprofit + nonprofit_amt,
          total_sales = total_sales + 1,
          total_revenue = total_revenue + NEW.total_amount
        WHERE id = referral_record.id;
        
      ELSIF referral_record.status = 'graduated' THEN
        -- Graduated: Split based on allocation preference
        IF referral_record.donation_allocation = 'sponsor' THEN
          sponsor_amt := donation_amt;
          nonprofit_amt := 0;
        ELSIF referral_record.donation_allocation = 'nonprofit' THEN
          sponsor_amt := 0;
          nonprofit_amt := donation_amt;
        ELSIF referral_record.donation_allocation = 'split' THEN
          sponsor_amt := donation_amt * ((referral_record.donation_split->>'sponsor')::DECIMAL / 100);
          nonprofit_amt := donation_amt * ((referral_record.donation_split->>'nonprofit')::DECIMAL / 100);
        END IF;
        
        -- Record sponsor donation
        IF sponsor_amt > 0 THEN
          INSERT INTO donation_transactions (
            transaction_id, mentee_id, recipient_type, recipient_id, amount, percentage
          ) VALUES (
            NEW.id, referral_record.mentee_id, 'sponsor', referral_record.sponsor_id, sponsor_amt, referral_record.donation_percentage
          );
          
          UPDATE member_referrals
          SET total_donated_to_sponsor = total_donated_to_sponsor + sponsor_amt
          WHERE id = referral_record.id;
          
          UPDATE auth.users
          SET total_mentorship_earnings = total_mentorship_earnings + sponsor_amt
          WHERE id = referral_record.sponsor_id;
        END IF;
        
        -- Record nonprofit donation
        IF nonprofit_amt > 0 THEN
          INSERT INTO donation_transactions (
            transaction_id, mentee_id, recipient_type, recipient_id, amount, percentage
          ) VALUES (
            NEW.id, referral_record.mentee_id, 'nonprofit', NULL, nonprofit_amt, referral_record.donation_percentage
          );
          
          UPDATE member_referrals
          SET total_donated_to_nonprofit = total_donated_to_nonprofit + nonprofit_amt
          WHERE id = referral_record.id;
        END IF;
      END IF;
    END IF;
  END IF;
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER after_transaction_complete
  AFTER INSERT OR UPDATE ON transactions
  FOR EACH ROW
  EXECUTE FUNCTION process_mentorship_donation();
```

### Graduation Check Function
```sql
-- Check if mentee is ready to graduate
CREATE OR REPLACE FUNCTION check_graduation_eligibility(mentee_id_input UUID)
RETURNS BOOLEAN AS $$
DECLARE
  referral_record RECORD;
  is_eligible BOOLEAN := false;
BEGIN
  SELECT * INTO referral_record
  FROM member_referrals mr
  JOIN users u ON u.id = mr.mentee_id
  WHERE mr.mentee_id = mentee_id_input
  AND mr.status = 'training';
  
  IF FOUND THEN
    -- Check graduation requirements
    IF referral_record.meetings_attended >= 4
       AND referral_record.total_sales >= 10
       AND referral_record.total_revenue >= 1000
       AND referral_record.avg_rating >= 4.5
       AND referral_record.sponsor_check_ins >= 3 THEN
      
      is_eligible := true;
      
      -- Auto-graduate
      UPDATE member_referrals
      SET 
        status = 'graduated',
        training_completed_at = NOW()
      WHERE mentee_id = mentee_id_input AND status = 'training';
      
      -- Award achievement
      INSERT INTO member_achievements (
        user_id, achievement_type, achievement_name, achievement_description
      ) VALUES (
        mentee_id_input, 'graduated', 'Graduated Training', 'Completed mentorship training program'
      ) ON CONFLICT (user_id, achievement_type) DO NOTHING;
      
      -- Update sponsor stats
      UPDATE auth.users
      SET successful_graduates = successful_graduates + 1
      WHERE id = referral_record.sponsor_id;
      
    END IF;
  END IF;
  
  RETURN is_eligible;
END;
$$ LANGUAGE plpgsql;
```
