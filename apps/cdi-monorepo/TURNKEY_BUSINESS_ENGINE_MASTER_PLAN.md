### Database Schema (Supabase):

```sql
-- Turnkey Businesses (created by nonprofit)
CREATE TABLE turnkey_businesses (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name TEXT NOT NULL,
  description TEXT,
  category TEXT, -- "Painters", "Mechanics", etc.
  city TEXT,
  state TEXT,
  llc_name TEXT,
  ein TEXT ENCRYPTED,
  business_plan_url TEXT,
  branding_package_url TEXT,
  supplier_contracts_url TEXT,
  created_by UUID REFERENCES auth.users(id),
  created_at TIMESTAMP DEFAULT NOW(),
  status TEXT DEFAULT 'draft' -- draft, auction, sold
);

-- Auctions
CREATE TABLE auctions (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  business_id UUID REFERENCES turnkey_businesses(id),
  starting_bid DECIMAL,
  current_bid DECIMAL,
  current_bidder UUID REFERENCES auth.users(id),
  start_date TIMESTAMP,
  end_date TIMESTAMP,
  status TEXT DEFAULT 'active', -- active, ended, cancelled
  created_at TIMESTAMP DEFAULT NOW()
);

-- Bids
CREATE TABLE bids (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  auction_id UUID REFERENCES auctions(id),
  bidder_id UUID REFERENCES auth.users(id),
  amount DECIMAL,
  timestamp TIMESTAMP DEFAULT NOW()
);

-- Idea Board (crowdfunding)
CREATE TABLE business_ideas (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  owner_id UUID REFERENCES auth.users(id),
  turnkey_business_id UUID REFERENCES turnkey_businesses(id), -- if from auction
  name TEXT NOT NULL,
  description TEXT,
  business_plan TEXT,
  funding_goal DECIMAL,
  current_funding DECIMAL DEFAULT 0,
  deadline TIMESTAMP,
  payment_account TEXT, -- CashApp/Stripe account
  nft_contract_address TEXT,
  nft_total_supply INTEGER,
  redemption_value DECIMAL,
  status TEXT DEFAULT 'active', -- active, funded, expired
  created_at TIMESTAMP DEFAULT NOW()
);

-- Supporters (crowdfunding backers)
CREATE TABLE idea_supporters (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  idea_id UUID REFERENCES business_ideas(id),
  supporter_id UUID REFERENCES auth.users(id),
  amount DECIMAL,
  nft_coins_promised INTEGER,
  nft_coins_received INTEGER DEFAULT 0,
  timestamp TIMESTAMP DEFAULT NOW()
);

-- NFT Collections (merchant coins)
CREATE TABLE nft_collections (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  business_id UUID REFERENCES business_ideas(id),
  contract_address TEXT,
  chain TEXT DEFAULT 'polygon',
  symbol TEXT,
  total_supply INTEGER,
  minted_count INTEGER DEFAULT 0,
  redemption_value DECIMAL,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Service Directory
CREATE TABLE directory_listings (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  business_id UUID REFERENCES business_ideas(id),
  category TEXT,
  city TEXT,
  state TEXT,
  rating DECIMAL,
  review_count INTEGER DEFAULT 0,
  accepts_nft_coins BOOLEAN DEFAULT true,
  accepts_time_banking BOOLEAN DEFAULT false,
  featured BOOLEAN DEFAULT false,
  created_at TIMESTAMP DEFAULT NOW()
);
```
